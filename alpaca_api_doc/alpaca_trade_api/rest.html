 118    @staticmethod
 119    def validate(amount: int, unit: TimeFrameUnit):
 120        if amount <= 0:
 121            raise ValueError("Amount must be a positive integer value.")
 122
 123        if unit == TimeFrameUnit.Minute and amount > 59:
 124            raise ValueError("Second or Minute units can only be " +
 125                             "used with amounts between 1-59.")
 126
 127        if unit == TimeFrameUnit.Hour and amount > 23:
 128            raise ValueError("Hour units can only be used with amounts 1-23")
 129
 130        if unit in (TimeFrameUnit.Day, TimeFrameUnit.Week) and amount != 1:
 131            raise ValueError(
 132                "Day and Week units can only be used with amount 1")
 133
 134        if unit == TimeFrameUnit.Month and amount not in (1, 2, 3, 6, 12):
 135            raise ValueError(
 136                "Month units can only be used with amount 1, 2, 3, 6 and 12")

 179    def _request(self,
 180                 method,
 181                 path,
 182                 data=None,
 183                 base_url: URL = None,
 184                 api_version: str = None):
 185        base_url = base_url or self._base_url
 186        version = api_version if api_version else self._api_version
 187        url: URL = URL(base_url + '/' + version + path)
 188        headers = {}
 189        if self._oauth:
 190            headers['Authorization'] = 'Bearer ' + self._oauth
 191        else:
 192            headers['APCA-API-KEY-ID'] = self._key_id
 193            headers['APCA-API-SECRET-KEY'] = self._secret_key
 194        headers['User-Agent'] = 'APCA-TRADE-SDK-PY/' + __version__
 195        opts = {
 196            'headers':         headers,
 197            # Since we allow users to set endpoint URL via env var,
 198            # human error to put non-SSL endpoint could exploit
 199            # uncanny issues in non-GET request redirecting http->https.
 200            # It's better to fail early if the URL isn't right.
 201            'allow_redirects': False,
 202        }
 203        if method.upper() in ['GET', 'DELETE']:
 204            opts['params'] = data
 205        else:
 206            opts['json'] = data
 207
 208        retry = self._retry
 209        if retry < 0:
 210            retry = 0
 211        while retry >= 0:
 212            try:
 213                return self._one_request(method, url, opts, retry)
 214            except RetryException:
 215                retry_wait = self._retry_wait
 216                logger.warning(
 217                    'sleep {} seconds and retrying {} '
 218                    '{} more time(s)...'.format(
 219                        retry_wait, url, retry))
 220                time.sleep(retry_wait)
 221                retry -= 1
 222                continue

 224    def _one_request(self, method: str, url: URL, opts: dict, retry: int):
 225        """
 226        Perform one request, possibly raising RetryException in the case
 227        the response is 429. Otherwise, if error text contain "code" string,
 228        then it decodes to json object and returns APIError.
 229        Returns the body json in the 200 status.
 230        """
 231        retry_codes = self._retry_codes
 232        resp = self._session.request(method, url, **opts)
 233        try:
 234            resp.raise_for_status()
 235        except HTTPError as http_error:
 236            # retry if we hit Rate Limit
 237            if resp.status_code in retry_codes and retry > 0:
 238                raise RetryException()
 239            if 'code' in resp.text:
 240                error = resp.json()
 241                if 'code' in error:
 242                    raise APIError(error, http_error)
 243            else:
 244                raise
 245        if resp.text != '':
 246            return resp.json()
 247        return None

 548    def _data_get(self,
 549                  endpoint: str,
 550                  symbol_or_symbols: Union[str, List[str]],
 551                  api_version: str = 'v2',
 552                  endpoint_base: str = 'stocks',
 553                  resp_grouped_by_symbol: Optional[bool] = None,
 554                  page_limit: int = DATA_V2_MAX_LIMIT,
 555                  **kwargs):
 556        page_token = None
 557        total_items = 0
 558        limit = kwargs.get('limit')
 559        if resp_grouped_by_symbol is None:
 560            resp_grouped_by_symbol = not isinstance(symbol_or_symbols, str)
 561        while True:
 562            actual_limit = None
 563            if limit:
 564                actual_limit = min(int(limit) - total_items, page_limit)
 565                if actual_limit < 1:
 566                    break
 567            data = kwargs
 568            data['limit'] = actual_limit
 569            data['page_token'] = page_token
 570            path = f'/{endpoint_base}'
 571            if isinstance(symbol_or_symbols, str) and symbol_or_symbols:
 572                path += f'/{symbol_or_symbols}'
 573            else:
 574                data['symbols'] = ','.join(symbol_or_symbols)
 575            if endpoint:
 576                path += f'/{endpoint}'
 577            resp = self.data_get(path, data=data, api_version=api_version)
 578            if not resp_grouped_by_symbol:
 579                k = endpoint or endpoint_base
 580                for item in resp.get(k, []) or []:
 581                    yield item
 582                    total_items += 1
 583            else:
 584                by_symbol = resp.get(endpoint, {}) or {}
 585                for sym, items in sorted(by_symbol.items()):
 586                    for item in items or []:
 587                        item['S'] = sym
 588                        yield item
 589                        total_items += 1
 590            page_token = resp.get('next_page_token')
 591            if not page_token:
 592                break

 225    def _one_request(self, method: str, url: URL, opts: dict, retry: int):
 226        """
 227        Perform one request, possibly raising RetryException in the case
 228        the response is 429. Otherwise, if error text contain "code" string,
 229        then it decodes to json object and returns APIError.
 230        Returns the body json in the 200 status.
 231        """
 232        retry_codes = self._retry_codes
 233        resp = self._session.request(method, url, **opts)
 234        try:
 235            resp.raise_for_status()
 236        except HTTPError as http_error:
 237            # retry if we hit Rate Limit
 238            if resp.status_code in retry_codes and retry > 0:
 239                raise RetryException()
 240            if 'code' in resp.text:
 241                error = resp.json()
 242                if 'code' in error:
 243                    raise APIError(error, http_error)
 244            else:
 245                raise
 246        if resp.text != '':
 247            return resp.json()
 248        return None
